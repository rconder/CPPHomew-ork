#### Aşağıdaki programda tanımsız davranış oluşturan kodları işaretleyin. Her bir tanımsız davranışın nedenini açıklayın:

```
#include <iostream>

class A {
public:
	void print()const
	{
		std::cout << mx << '\n';
	}
private:
	int mx;
};

A ga;

int main()
{
	A a1;
	A a2{};
	static A a3;

	ga.print();
	a1.print();
	a2.print();
	a3.print();
}

```
