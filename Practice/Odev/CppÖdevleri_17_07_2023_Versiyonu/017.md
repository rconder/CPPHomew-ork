#include <iostream>

int main()
{
	int x = 10;
	int *ptr = &x;

	auto r1 = x;//tür int
	auto r2 = *ptr;//*ptr türünün ifadesi int 
	auto r3 = r2;//int 
	auto &r4 = ptr;//int * & 
	auto &r5 = *ptr;//int türden bir nesneye referans

	r1 += 3;
	r2 += 13;
	r3 += 20;
	*r4 += r2;//r4 demek ptr demek *r4 demek x demek olduğu için x'in değeri 33 olucak.
	++r5;//34 olucak

	std::cout << "x = " << x << "\n";

}
